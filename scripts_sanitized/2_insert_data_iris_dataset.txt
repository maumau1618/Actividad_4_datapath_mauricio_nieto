-- Script SQL para cargar toda la data Iris en tu tabla
-- Fuente: dataset publico de BigQuery

INSERT INTO `TU_PROJECT_ID.TU_DATASET.iris_data` (
	id,
	sepal_length_cm,
	sepal_width_cm,
	petal_length_cm,
	petal_width_cm,
	species
)
SELECT
	ROW_NUMBER() OVER (ORDER BY species, sepal_length, sepal_width, petal_length, petal_width) AS id,
	CAST(sepal_length AS FLOAT64) AS sepal_length_cm,
	CAST(sepal_width AS FLOAT64) AS sepal_width_cm,
	CAST(petal_length AS FLOAT64) AS petal_length_cm,
	CAST(petal_width AS FLOAT64) AS petal_width_cm,
	CAST(species AS STRING) AS species
FROM `bigquery-public-data.ml_datasets.iris`;
